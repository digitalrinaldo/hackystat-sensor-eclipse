001    package org.hackystat.sensor.eclipse;
002    
003    import java.util.MissingResourceException;
004    import java.util.ResourceBundle;
005    
006    /**
007     * Provides internationalization resources from resource bundle.
008     *
009     * @author Takuya Yamashita
010     * @version $Id: EclipseSensorI18n.java,v 1.1.1.1 2005/10/20 23:56:56 johnson Exp $
011     */
012    public class EclipseSensorI18n {
013      /** the resource bundle file name */
014      private static final String RESOURCE_BUNDLE = "org.hackystat.sensor.eclipse.Resources";
015      /** the resource bundle instance. */
016      private static ResourceBundle bundle = ResourceBundle.getBundle(RESOURCE_BUNDLE);
017    
018      /**
019       * Prevents clients from instantiating this class.
020       */
021      private EclipseSensorI18n() {
022      }
023      
024      /**
025       * Returns the string from the plugin's resource bundle,
026       * or 'key' if not found.
027       * @param key the key to search value in the resource bundle.
028       * @return the string from the plugin's resource bundle,
029       * or 'key' if not found.
030       */
031      public static String getString(String key) {
032        try {
033          return bundle.getString(key);
034        } 
035        catch (MissingResourceException e) {
036          return key;
037        }
038      }
039    }
