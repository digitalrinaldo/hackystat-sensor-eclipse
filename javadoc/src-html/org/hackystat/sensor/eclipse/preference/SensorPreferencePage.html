001    package org.hackystat.sensor.eclipse.preference;
002    
003    import org.eclipse.jface.preference.FieldEditorPreferencePage;
004    import org.eclipse.jface.preference.IntegerFieldEditor;
005    import org.eclipse.jface.preference.StringFieldEditor;
006    import org.eclipse.ui.IWorkbench;
007    import org.eclipse.ui.IWorkbenchPreferencePage;
008    import org.hackystat.sensor.eclipse.EclipseSensor;
009    import org.hackystat.sensor.eclipse.EclipseSensorPlugin;
010    import org.hackystat.sensorbase.client.SensorBaseClient;
011    import org.hackystat.sensorshell.SensorShellException;
012    
013    /**
014     * Implements the preference page for Eclipse Sensor. It does some validation to make sure
015     * that hackystat sensorbase host, email and password are correct. 
016     * 
017     * @author Hongbing Kou
018     */
019    public class SensorPreferencePage extends FieldEditorPreferencePage 
020        implements IWorkbenchPreferencePage {
021      /** SensorBase field. */
022      private StringFieldEditor sensorbaseField; 
023      /** Email field. */
024      private StringFieldEditor emailField;
025      /** Password field. */
026      private StringFieldEditor passwordField;
027      /** Autosend Interval field. */
028      private IntegerFieldEditor autosendIntervalField;
029      
030      /** Autoupdate field. */
031      //private BooleanFieldEditor autoUpdateEditor;
032      /** Update site. */
033      //private StringFieldEditor udpateSiteField;
034      
035      /** 
036       * Instantiate Eclipse plugin preference.
037       */
038      public SensorPreferencePage() {
039        super(GRID);
040        
041        EclipseSensorPlugin plugin = EclipseSensorPlugin.getDefault();
042        setPreferenceStore(plugin.getPreferenceStore());
043      }
044      
045      /** Creates the field editors. Field editors are abstractions of
046       * the common GUI blocks needed to manipulate various types
047       * of preferences. Each field editor knows how to save and
048       * restore itself.
049       */
050      public void createFieldEditors() {
051        // hackystat sensorbase host
052        this.sensorbaseField = new StringFieldEditor(
053            PreferenceConstants.P_SENSORBASE, "&SensorBase", getFieldEditorParent());
054        this.sensorbaseField.setEmptyStringAllowed(false);
055        super.addField(this.sensorbaseField);
056    
057        // User email
058        this.emailField = new StringFieldEditor(
059            PreferenceConstants.P_EMAIL, "&Email", getFieldEditorParent());
060        this.emailField.setEmptyStringAllowed(false);
061        super.addField(this.emailField);
062    
063        // Password
064        this.passwordField = new StringFieldEditor(
065            PreferenceConstants.P_PASSWORD, "&Password", getFieldEditorParent());
066        this.passwordField.setEmptyStringAllowed(false);
067        this.passwordField.getTextControl(getFieldEditorParent()).setEchoChar('*');
068        super.addField(this.passwordField);
069    
070        this.autosendIntervalField = new IntegerFieldEditor(PreferenceConstants.P_AUTOSEND_INTERVAL, 
071            "&Autosend Interval (Minutes)", getFieldEditorParent());
072        this.autosendIntervalField.setEmptyStringAllowed(false);
073        super.addField(this.autosendIntervalField);
074    
075        // Whether or not allow auto update
076    //    this.autoUpdateEditor = new BooleanFieldEditor(
077    //        PreferenceConstants.P_ENABLE_AUTOUPDATE,
078    //        "Enable &AutoUpdate", getFieldEditorParent());
079    //    super.addField(this.autoUpdateEditor);
080        
081    //    this.udpateSiteField = new StringFieldEditor(
082    //        PreferenceConstants.P_UPDATE_SITE, "&Update Site", getFieldEditorParent());
083    //    super.addField(this.udpateSiteField);
084      }
085    
086      /**
087       * Initialize the preference page. 
088       * 
089       * @param workbench Eclipse workbench.
090       */
091      public void init(IWorkbench workbench) {
092        //Initialize the preference store we wish to use
093        setPreferenceStore(EclipseSensorPlugin.getDefault().getPreferenceStore());
094      }
095      
096      /**
097       * Perform okay to validate sensorbase/user/password.
098       * 
099       * @return True if inputs are valid.
100       */
101      public boolean performOk() {
102        //IPreferenceStore store = super.getPreferenceStore();
103        String sensorbase = this.sensorbaseField.getStringValue();
104        if (!SensorBaseClient.isHost(sensorbase)) {
105          super.setErrorMessage(sensorbase + " is an invalid hackystat host!");
106          return false;
107        }
108        
109        String email = this.emailField.getStringValue(); 
110        //store.getString(PreferenceConstants.P_EMAIL);
111        String password = this.passwordField.getStringValue();
112        //store.getDefaultString(PreferenceConstants.P_PASSWORD);
113        if (!SensorBaseClient.isRegistered(sensorbase, email, password)) {
114          super.setErrorMessage("Either email or password is incorrect!");
115          return false;
116        }
117        
118       // Ask sensor to enable the new settings.
119        try {
120          super.performOk();
121    
122          EclipseSensor sensor = EclipseSensor.getInstance();
123          // Stop the sensor temporarily to let it send out previously collected sensor data.
124          sensor.stop();
125          sensor.hotDeploySensorShell();
126        }
127        catch (SensorShellException e) {
128          super.setErrorMessage(e.getMessage());
129          return false;
130        }
131        
132        return true;
133      }
134    }
