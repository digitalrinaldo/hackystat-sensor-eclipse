001    package org.hackystat.sensor.eclipse.event;
002    
003    
004    /**
005     * Provides an even listener for the information Eclipse sensor gathers. A client 
006     * must implement this class and override the notify method in order to be notified 
007     * when a event of action (such as opened
008     * project, closed project, saved file, and so forth) is triggered. the client can know
009     * what kind of even is triggered, checking the even flag of the EclipseSensorEvent instance.
010     * For, example, to check if the event is notified when project is opened, the client 
011     * can override the notify method such as:
012     * 
013     * <pre>
014     * public void notify (EclispeSensorEvent event) {
015     *   if (event.getEvent() != EclipseSensorEvent.PROJECT_OPEN) {
016     *     return;
017     *   }
018     *   String workspacePath = event.getResource().getWorkspace().getRoot().getLocation().toString();
019     *   IFolder cvsFolder = ((IProject) event.getResource()).getFolder("CVS");
020     *   ...
021     * }
022     * </pre>
023     * 
024     * After implementing the sub class, the client need to add this class to the 
025     * <code>org.hackystat.sensor.eclipse.EclipseSensor</code>. To add the implementing class to
026     * the <code>EclipseSensor</code>, use <code>EclipseSensor#addEclipseSensorEventListener()</code>
027     * method such as:
028     * 
029     * <pre>
030     * EclipseSensor.getInstance().addEclipseSensorEventListener(this);
031     * </pre>
032     * 
033     * Note that the "this" points to the implementing sub class.
034     * <p>Whenever a new sensor properties is supposed to be read in Eclipse sensor, you must set the 
035     * new sensor properties in such a way that :
036     * 
037     * <pre>
038     * File sensorPropertiesFile = new File(...);
039     * SensorProperties newProperties = new SensorProperties(EclipseSensor.EclipseSensor.SENSOR_ECLIPSE,
040     *                                                       sensorPropertiesFile);
041     * EclipseSensor.getInstance().setSensorProperties(newProperties);
042     * </pre>
043     *
044     * @author Takuya Yamashita
045     * @version $Id: IEclipseSensorEventListener.java,v 1.1.1.1 2005/10/20 23:56:57 johnson Exp $
046     */
047    public interface IEclipseSensorEventListener {
048      /**
049       * Provides a notification method to be called when the implementing class is notified.
050       *
051       * @param event The EclipseSensorEvent instance to hold an event information (flag).
052       */
053      void notify(EclipseSensorEvent event);
054    }
