001    package org.hackystat.sensor.eclipse.event;
002    
003    import org.eclipse.core.resources.IResource;
004    
005    
006    /**
007     * Provides a event information (flag) and IResource instance if applicable. A client
008     * of the Eclipse Sensor should not instantiate this instance. IResource instance is given at 
009     * the resource place where the EclipseSensorEventListener is hooked.
010     * 
011     * <p>The IResource instance can be used when the following flag is contained.
012     * <ul>
013     * <li>PROJECT_OPEN - A project is opened. IResource can cast IProject.</li>
014     * <li>PROJECT_CLOSE - A project is closed. IResource can cast IProject.</li>
015     * <li>FILE_SAVE - A file is saved. IResource can cast IFile.</li>
016     * 
017     * After getting IResource instance, using <code>getResource()</code>, the client
018     * might manipulate the project path and/or workspace path, and so forth.
019     * 
020     * <p>Otherwise the IResource instance is null.
021     *
022     * @author Takuya Yamashita
023     * @version $Id: EclipseSensorEvent.java,v 1.1.1.1 2005/10/20 23:56:57 johnson Exp $
024     * @see org.eclipse.core.resources.IResource
025     */
026    public class EclipseSensorEvent {
027      /**
028       * The flag when an event notify that project is opened.
029       */
030      public static final int PROJECT_OPEN = 0;
031    
032    
033      /**
034       * The flag when an event notify that project is closed.
035       */
036      public static final int PROJECT_CLOSE = 1;
037    
038    
039      /**
040       * The flag when an event notify that a file is saved.
041       */
042      public static final int FILE_SAVE = 2;
043    
044    
045      /**
046       * The IResource interface to hold either IContainer, IFile, IFolder, IProject, or IWorkspaceRoot.
047       */
048      private IResource resource;
049    
050    
051      /**
052       * The event information (flag). For example, that project is opened is 0.
053       */
054      private int event;
055    
056    
057      /**
058       * Constructor for the EclipseSensorEvent object. A client of the Eclipse sensor should
059       * not instantiate this instance. Only EclipseSenserPlugin can instantiate this.
060       *
061       * @param resource IResource where the EclipseSensorEventListener is hooked.
062       * @param event The event information when the EclipseSensor notify.
063       */
064      public EclipseSensorEvent(IResource resource, int event) {
065        this.resource = resource;
066        this.event = event;
067      }
068    
069    
070      /**
071       * Gets the IResource implementing interface or class. After getting this. a client can cast
072       * the IResource sub interface or sub class such as IProject, IFile, and so forth.
073       *
074       * @return The IResource interface.
075       */
076      public IResource getResource() {
077        return this.resource;
078      }
079    
080    
081      /**
082       * Gets the event information (flag). A client can know the event kind in the 
083       * EclipseSensorEventListener implementing class when the class is notified.
084       *
085       * @return The event information (flag).
086       */
087      public int getEvent() {
088        return this.event;
089      }
090    }
