001    package org.hackystat.sensor.eclipse;
002    
003    import java.util.Map;
004    
005    import org.eclipse.jface.action.IStatusLineManager;
006    import org.eclipse.swt.widgets.Display;
007    import org.eclipse.ui.IActionBars;
008    import org.eclipse.ui.IEditorPart;
009    import org.eclipse.ui.IViewPart;
010    import org.eclipse.ui.IViewReference;
011    import org.eclipse.ui.IWorkbench;
012    import org.eclipse.ui.IWorkbenchPage;
013    import org.eclipse.ui.IWorkbenchWindow;
014    import org.hackystat.sensorshell.SensorShell;
015    import org.hackystat.sensorshell.SensorShellException;
016    import org.hackystat.sensorshell.SensorShellProperties;
017    
018    /**
019     * Provides the sensor shell wrapper class, which provides eclipse specific invocation such that 
020     * the collected sensor data are shown on the status bar for monitoring. 
021     * 
022     * @author Hongbing Kou
023     */
024    public class SensorShellWrapper {
025      /** The sensor shell instance. */
026      private SensorShell shell;
027    
028      /**
029       * Instantiates the eclipse specific sensor shell.
030       * 
031       * @param sensorShellProperties the SensorProperties instance.
032       * 
033       * @throws SensorShellException If error occurs in instantiating the sensorshell.
034       */
035      SensorShellWrapper(SensorShellProperties sensorShellProperties) throws SensorShellException {
036        this.shell = new SensorShell(sensorShellProperties, false, "Eclipse");
037      }
038      
039      /**
040       * Add key-value pairs of metrics to sensorshell to be sent to the Hackystat server 
041       * automatically.
042       * 
043       * @param keyValuePairs Key-value pairs of metrics.
044       * @param message A message to appear on the Eclipse's status bar. 
045       */
046      public void add(Map<String, String> keyValuePairs, String message) {
047        processStatusLine("Hackystat Sensor : " + message);
048     
049        try {
050          this.shell.add(keyValuePairs);
051        }
052        catch (Exception e) {
053          EclipseSensorPlugin plugin = EclipseSensorPlugin.getDefault();
054          plugin.log(e);
055          processStatusLine("Hackystat Sensor : Error occurred when sending data to server. ");
056        }
057      }
058      
059      /**
060       * Sends all Hackystat data to the server. Do nothing if sensor shell instance is null.
061       */
062      public void send() {
063        try {
064          this.shell.send();
065        }
066        catch (SensorShellException e) {
067          EclipseSensorPlugin plugin = EclipseSensorPlugin.getDefault();
068          plugin.log(e);
069          processStatusLine("Hackystat Sensor : Error occurred when sending data to server. ");
070        }
071      }
072      
073      /**
074       * Quick sensorshell when Hackystat Eclipse Sensor is closed or stopped. 
075       * 
076       */
077      public void quit() {
078        this.shell.quit();
079      }
080    
081      /**
082       * Processes to display the message in the status line. Since EclipseSensor is executed
083       * from a non-UI thread, such the application that wishs to call UI code from the non-UI thread
084       * must provide a Runnable (anonymous class in this case) that calls the UI code. 
085       * For more detail, try to search "Threading issues" in the "Help | Help Contents" of
086       * the Eclipse IDE.
087       * 
088       * @param message the message to be displayed.
089       */
090      private void processStatusLine(final String message) {
091        Display.getDefault().asyncExec(new Runnable() {
092    
093          public void run() {
094            // Retrieves status line manager instance and sets the message into the instance.
095            IWorkbench workbench = EclipseSensorPlugin.getInstance().getWorkbench();
096            IWorkbenchWindow activeWindow = workbench.getActiveWorkbenchWindow();
097            if (activeWindow != null) {
098              IWorkbenchPage activePage = activeWindow.getActivePage();
099              if (activePage != null) {
100                showStatusMessage(activePage);
101              }
102            }
103          }
104          
105          /**
106           * Show the status message.
107           *  
108           * @param activePage The active workbench.
109           */
110          private void showStatusMessage(IWorkbenchPage activePage) {
111            IViewReference[] viewReferences = activePage.getViewReferences();
112            for (int i = 0; i < viewReferences.length; i++) {
113              IViewPart viewPart = viewReferences[i].getView(true);
114              if (viewPart != null) {
115                IActionBars viewActionbars = viewPart.getViewSite().getActionBars();
116                IStatusLineManager viewStatusManager = viewActionbars.getStatusLineManager();
117                // Set status line associated with view part such as Package Explore, and etc.
118                viewStatusManager.setMessage(message);
119              }
120            }
121            IEditorPart editorPart = activePage.getActiveEditor();
122            if (editorPart != null) {
123              IActionBars partActionBars = editorPart.getEditorSite().getActionBars();
124              IStatusLineManager partStatusManager = partActionBars.getStatusLineManager();
125              // Set status line associated with editor part such as text editor.
126              partStatusManager.setMessage(message);
127            }
128    
129          }
130        });
131      }
132    }
