027    /**
028     * Provides the version check function. Clients may call <code>processUpdateDialog(String)</code>
029     * to check new version on the server, whose url is specified in the preference page, and
030     * bring update info dialog if the new version is available. Supoorts for both 2x and 3x stream
031     * version information.
032     * 
033     * @author Hongbing Kou
034     */
035    public class VersionCheck {
036      /** Bundle object. */
037      private Bundle bundle;
038      
039      /**
040       * Instantiates a version check to a given bundle.
041       * 
042       * @param bundle A resource bundle or plugin.
043       */
044      public VersionCheck(Bundle bundle) {
045        this.bundle = bundle;
046      }
047      
048      /**
049       * Creates an image descriptor form the given path. The path should be the relative path from a
050       * project root.
051       *
052       * @param path the relative path from a project root.
053       *
054       * @return the newly created image descriptor.
055       */
056      public ImageDescriptor createImageDescriptor(String path) {
057        try {
058          URL url = this.bundle.getEntry("/");
059          return ImageDescriptor.createFromURL(new URL(url, path));
060        }
061        catch (MalformedURLException e) {
062          return ImageDescriptor.getMissingImageDescriptor();
063        }
064        catch (Exception e) {
065          e.printStackTrace();
066          return ImageDescriptor.getMissingImageDescriptor();
067        }
068      }
069      
070      /**
071       * Processes update dialog to the user. The current version and new version appear in the dialog.
072       * The element of the messages array is follows:
073       * <ul>
074       * <li>The element 0: first message.
075       * <li>The element 1: the message between local version and server version.
076       * <li>The element 2: last message.
077       * </ul>  
078       * @param updateUrl the url for the update site, which ends with "site.xml".
079       * @param title the title to be shown in the update window page.
080       * @param messages the messages to be shown in the update window page. 
081       */
082      public void processUpdateDialog(String updateUrl, final String title, final String[] messages) {
083        if (updateUrl == null) {
084          return;
085        }
086        
087        try { // Gets current version info.
088          Version localVerIdentifier = getLocalPluginVersionIdentifier();
089          String localVerString = localVerIdentifier.toString();
090          final String localVersionId = localVerString.substring(0, localVerString.length() - 2);
091          final String qualifierVersion = localVerIdentifier.getQualifier();
092    
093          // Gets new version info.
094          URL url = new URL(updateUrl);
095          Document serverDocument = parseXml(url.openStream());
096          final Version serverVerIdentifier = getVersionIdentifier(serverDocument, 
097              qualifierVersion);
098          // Check if current version is not new version, then show pop up udate dialog.
099          if (serverVerIdentifier != null && serverVerIdentifier.compareTo(localVerIdentifier) > 0) {
100            String serverVerString = serverVerIdentifier.toString();
101            final String  serverVerId = serverVerString.substring(0, serverVerString.length() - 2);
102            
103            IWorkbench workbench = EclipseSensorPlugin.getInstance().getWorkbench();
104            final IWorkbenchWindow window = workbench.getWorkbenchWindows()[0];
105            final String HACKYSTAT_ICON = EclipseSensorI18n.getString("VersionCheck.hackystatIcon");
106            Display.getDefault().asyncExec(new Runnable() {
107              public void run() {
108                MessageDialog dialog = new MessageDialog(window.getShell(),
109                title,
110                createImageDescriptor(HACKYSTAT_ICON).createImage(), 
111                messages[0] + localVersionId + messages[1] + serverVerId + 
112                  messages[2],
113                MessageDialog.QUESTION,
114                new String[] {
115                  EclipseSensorI18n.getString("VersionCheck.messageDialogButtonUpdate"),
116                  EclipseSensorI18n.getString("VersionCheck.messageDialogButtonCancel")
117                },
118                0);
119                  
120                int result = dialog.open();
121                if (result == 0) {
122                  openNewUpdatesWizard();
123                }
124              }
125             });
126          }
127        }
128        catch (Exception e) {
129          e.printStackTrace();
130          EclipseSensorPlugin.getInstance().log(e);
131          //couldn't parse xml file, or even though xml is parsed, version attribute is not found.
132        }
133      }
134      
135      /**
136       * Gets the local plugin version identifier.
137       * 
138       * @return the local plugin version identifier.
139       */
140      private Version getLocalPluginVersionIdentifier() {
141        String version = "8.0.0";
142        if (this.bundle != null) {
143          version = (String) this.bundle.getHeaders().get(Constants.BUNDLE_VERSION);  
144        }
145        
146        return new Version(version);    
147      }
148    
149      /**
150       * Gets version information from the given Document object, checking the qualifier version. The
151       * qualifier version is the last token of the version identifier. For example, if the version
152       * identifier is 1.4.204.2x, the qualifier version is 2x.
153       *
154       * @param document The Document object parsed from XML file.
155       * @param qualifierVersion The qualifier version. e.g 2x for the 1.4.204.2x
156       *
157       * @return The PluginVersionIdentifier instance if it's found.Returns null if the version
158       *         information is not found.
159       */
160      private static Version getVersionIdentifier(Document document,
161        String qualifierVersion) {
162        Element root = document.getDocumentElement();
163        NodeList list = root.getChildNodes();
164        final String FEATURE = "feature";
165        for (int i = 0; i < list.getLength(); i++) {
166          if ((list.item(i).getNodeType() == Node.ELEMENT_NODE)
167              && list.item(i).getNodeName().equalsIgnoreCase(FEATURE)) {
168            Element element = (Element) list.item(i);
169            Version identifier = getVersion(element);
170            if (identifier != null && identifier.getQualifier().equals(qualifierVersion)) {
171              return identifier;
172            }
173          }
174        }
175        return null;
176      }
177    
178      /**
179       * Gets the version from the element.
180       * 
181       * @param element Element node in the XML file.
182       * @return Version if the feature node has a valid version number. 
183       */
184      private static Version getVersion(Element element) {
185        Version identifier = null; 
186        
187        if (element.hasAttribute("version")) {
188          String version = element.getAttribute("version");
189          String[] versions = version.split("\\.");
190          String newVersion = versions[0] + "." + versions[1];   
191          identifier = new Version(newVersion);
192        }
193        
194        return identifier;
195      }

197      /**
198       * Parses xml file to generate Document object from a given input stream.
199       *
200       * @param input The given input stream from which xml file is read.
201       *
202       * @return The Documentation object which contains the content parsed from the XML file. Returns
203       *         null if parse error occurs.
204       */
205      private static Document parseXml(InputStream input) {
206        try {
207          // Create a builder factory
208          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
209          factory.setValidating(false);
210          // Create the builder and parse the file
211          Document document = factory.newDocumentBuilder().parse(input);
212          input.close();
213          return document;
214        }
215        catch (SAXException e) {
216          // A parsing error occurred; the xml input is not valid
217          EclipseSensorPlugin.getInstance().log(e);
218        }
219        catch (ParserConfigurationException e) {
220          EclipseSensorPlugin.getInstance().log(e);
221        }
222        catch (IOException e) {
223          EclipseSensorPlugin.getInstance().log(e);
224        }
225        return null;
226      }
