018    /**
019     * Listens to the java element change events to get incremental work on java objects and collect 
020     * refactoring information for test-driven development analysis purpose. It's declared as package
021     * private so that it can only be instantiated by Eclise sensor.
022     * 
023     * @author Hongbing Kou
024     * @version $Id$
025     */
026    public class JavaStructureChangeDetector implements IElementChangedListener {
027      /** Eclipse sensor which is used to send out hackystat data. */
028      private EclipseSensor sensor;
029    
030      /**
031       * Instantiates the JavaStructureDetector instance with Eclipse sensor.
032       * 
033       * @param sensor Eclipse sensor.
034       */
035      public JavaStructureChangeDetector(EclipseSensor sensor) {
036        this.sensor = sensor;
037      }
038    
039      /**
040       * Implements the element change response.
041       * 
042       * @param event Element change event.
043       */
044      public void elementChanged(ElementChangedEvent event) {
045        //IJavaElementDelta jed = event.getDelta().getAffectedChildren()[0];
046        IJavaElementDelta[] childrenChanges = event.getDelta().getAffectedChildren(); 
047        
048        if (childrenChanges != null && childrenChanges.length > 0) {
049          javaObjectChange(childrenChanges[0]);
050        }
051      }
052      
053      /**
054       * Process the editng on java element changes.
055       * 
056       * @param jed Java element delta change.
057       */
058      private void javaObjectChange(IJavaElementDelta jed) {
059        List<IJavaElementDelta> additions = new ArrayList<IJavaElementDelta>();
060        List<IJavaElementDelta> deletions = new ArrayList<IJavaElementDelta>();
061        
062        // Traverse the delta change tree for refactoring activity
063        traverse(jed, additions, deletions);
064        
065        //  Gets the location of java file.
066        IPath javaFile = jed.getElement().getResource().getLocation();
067            
068        // No java structure change
069        if (additions.isEmpty() && deletions.isEmpty()) {
070          return;      
071        }
072        // Addition, deletion, renaming activity.
073        else if (additions.size() == 1 || deletions.size() == 1) {
074          if (deletions.isEmpty()) {
075            processUnary(javaFile, "Add", (IJavaElementDelta) additions.get(0));        
076          }
077          else if (additions.isEmpty()) {
078            processUnary(javaFile, "Remove", (IJavaElementDelta) deletions.get(0));
079          }
080          else if (deletions.size() == 1) {
081            IJavaElementDelta fromDelta = (IJavaElementDelta) deletions.get(0);
082            IJavaElementDelta toDelta = (IJavaElementDelta) additions.get(0);
083            if (fromDelta.getElement().getParent().equals(toDelta.getElement().getParent())) { 
084              processRenameRefactor(javaFile, fromDelta, toDelta); 
085            }
086            else {
087              javaFile = fromDelta.getElement().getResource().getLocation();
088              processMoveRefactor(javaFile, fromDelta.getElement(), fromDelta.getElement().getParent(), 
089                  toDelta.getElement().getParent());
090            }
091          }
092        }    
093        // Massive addition by copying
094        else if (additions.size() > 1) {
095          for (Iterator i = additions.iterator(); i.hasNext();) {
096            processUnary(javaFile, "Add", (IJavaElementDelta) i.next());
097          }
098        }
099        // Massive block deletion
100        else if (deletions.size() > 1) {
101          for (Iterator i = deletions.iterator(); i.hasNext();) {
102            processUnary(javaFile, "Remove", (IJavaElementDelta) i.next());
103          }
104        }    
105      }

107      /**
108       * Constructs and sends the java element change data.
109       * 
110       * @param javaFile Associated file.
111       * @param op Operation
112       * @param delta Delta change element
113       */
114      private void processUnary(IPath javaFile, String op, IJavaElementDelta delta) {
115        IJavaElement element = delta.getElement();
116    
117        // Stop if there is no associated element. 
118        if (javaFile == null || element == null || element.getResource() == null) {
119          return;
120        }
121        
122        String type = retrieveType(element);
123        // If type is not field, method, import and class do nothing.
124        if (type == null) {
125          return;  
126        }
127        
128        IPath classFileName = javaFile;
129        if (EclipseSensorConstants.CLASS.equals(type)) {
130          classFileName = element.getResource().getLocation();
131        }  
132        
133        // Only deal with java file.
134        if (!EclipseSensorConstants.JAVA.equals(classFileName.getFileExtension())) {
135          return;
136        }
137        
138        String name = retrieveName(element);
139        //String toName = 
140        if (name != null && !"".equals(name)) {
141          Map<String, String> devEventPMap = new HashMap<String, String>();
142          devEventPMap.put("Subtype", "ProgramUnit");
143          devEventPMap.put("Subsubtype", op);
144          devEventPMap.put("Language", EclipseSensorConstants.JAVA);
145          devEventPMap.put("Unit-Type", type);
146          devEventPMap.put("Unit-Name", name);
147          
148          URI resource = element.getResource().getLocationURI();
149          this.sensor.addDevEvent("Edit", resource, devEventPMap, op + " " + name);
150        }
151      }

154      /**
155       * Constructs and send of the java element change data.
156       * 
157       * @param javaFile Associated file.
158       * @param fromDelta Change from delta.
159       * @param toDelta Change to delta.
160       */
161      private void processRenameRefactor(IPath javaFile, IJavaElementDelta fromDelta, 
162          IJavaElementDelta toDelta) {
163        String typeName = retrieveType(toDelta.getElement());
164        
165        IPath classFileName = javaFile;
166        if (EclipseSensorConstants.CLASS.equals(typeName)) {
167          classFileName = fromDelta.getElement().getResource().getLocation();
168        }
169        else if ("Package".equals(typeName)) {
170          classFileName = fromDelta.getElement().getResource().getLocation();
171        }
172    
173        // Only deal with java file.
174        if (!EclipseSensorConstants.JAVA.equals(classFileName.getFileExtension())) {
175          return;
176        }
177        
178        String fromName = retrieveName(fromDelta.getElement());
179        String toName = retrieveName(toDelta.getElement());
180        
181        if (fromName != null && !"".equals(fromName) && toName != null && !"".equals(toName)) {
182          StringBuffer msgBuf = new StringBuffer();
183          msgBuf.append("Refactor : Rename#").append(typeName).append('#').append(fromName)
184                .append(" -> ").append(toName);
185    
186          Map<String, String> devEventPMap = new HashMap<String, String>();
187          devEventPMap.put("Subtype", "ProgramUnit");
188          devEventPMap.put("Subsubtype", "Rename");
189          devEventPMap.put("Language", EclipseSensorConstants.JAVA);
190          devEventPMap.put("Unit-Type", typeName);
191          devEventPMap.put("From-Unit-Name", fromName);
192          devEventPMap.put("To-Unit-Name", toName);
193    
194          URI resource = fromDelta.getElement().getResource().getLocationURI();
195          this.sensor.addDevEvent("Edit", resource, devEventPMap, msgBuf.toString());
196        }
197      }
198      
199      /**
200       * Constructs and send of the java element change data.
201       * 
202       * @param javaFile Associated file.
203       * @param element  Java Element to be moved.
204       * @param from Change from element.
205       * @param to Change to element.
206       */
207      private void processMoveRefactor(IPath javaFile, IJavaElement element, 
208          IJavaElement from, IJavaElement to) {
209        String typeName = retrieveType(element);
210        
211        // Only deal with java file.
212        if (!EclipseSensorConstants.JAVA.equals(javaFile.getFileExtension())) {
213          return;
214        }
215        
216        String name = retrieveName(element);
217        String fromName = retrieveName(from);
218        String toName = retrieveName(to);
219        
220        // Put refactor data together with pound sigh separation and send it to Hackystat 
221        // server as activity data.
222        if (fromName != null && !"".equals(fromName) && toName != null && !"".equals(toName)) {
223          StringBuffer msgBuf = new StringBuffer();
224          msgBuf.append("Refactor : Move#").append(typeName).append('#').append(name).append('#').
225                 append(fromName).append(" -> ").append(toName);
226          
227          Map<String, String> devEventRenameMap = new HashMap<String, String>();
228          devEventRenameMap.put("Subtype", "ProgramUnit");
229          devEventRenameMap.put("Subsubtype", "Move");
230          devEventRenameMap.put("Language", EclipseSensorConstants.JAVA);
231          devEventRenameMap.put("Unit-Type", typeName);
232          // return-type not available
233          devEventRenameMap.put("From-Unit-Name", fromName);
234          devEventRenameMap.put("To-Unit-Name", toName);
235          
236          URI resource = element.getResource().getLocationURI();
237          this.sensor.addDevEvent("Edit", resource, devEventRenameMap, msgBuf.toString());
238        }    
239      }

241      /**
242       * Gets the element type.
243       * 
244       * @param element Java element object
245       * @return Element type string (class, method, field or import).
246       */
247      private String retrieveType(IJavaElement element) {
248        int eType = element.getElementType();
249        
250        switch (eType) {
251          case IJavaElement.FIELD:
252            return "Field";
253          case IJavaElement.METHOD:
254            return "Method";
255          case IJavaElement.IMPORT_DECLARATION:
256            return "Import";
257                case IJavaElement.IMPORT_CONTAINER:
258            return "Import";
259          case IJavaElement.COMPILATION_UNIT:
260            return EclipseSensorConstants.CLASS;
261          case IJavaElement.JAVA_PROJECT:
262            return EclipseSensorConstants.CLASS;
263          case IJavaElement.PACKAGE_FRAGMENT:
264            return "Package";
265          default:
266            return null;
267        }
268      }
269      
270      /**
271       * Gets the element name with signature.
272       * 
273       * @param element Java element, which could be class, method, field or import.
274       * @return Brief element name.
275       */
276      private String retrieveName(IJavaElement element) {
277        String name = element.toString();
278        try {
279          name = name.substring(0, name.indexOf('['));
280        }
281        catch (IndexOutOfBoundsException e) {
282          System.out.println("Where is the [ ? " + name);
283        }
284        // Trim off the meaningless "(not open)" string
285        int pos = name.indexOf("(not open)");
286        if (pos > 0) {
287          name = name.substring(0, pos);
288        }
289        
290        // take off the '#' if it exists
291        name = name.replace('#', '/');
292        
293        return name.trim(); 
294      }

296      /**
297       * Traverses the delta change tree on java element to look for addition and deletion on
298       * java element.
299       * 
300       * @param delta Delta element change.
301       * @param additions Added element holder.
302       * @param deletions Deleted element holder.
303       */
304      private void traverse(IJavaElementDelta delta, List<IJavaElementDelta> additions, 
305          List<IJavaElementDelta> deletions) {    
306        // Saves the addition and deletion.
307        if (delta.getKind() == IJavaElementDelta.ADDED) {
308           additions.add(delta);
309        }
310        else if (delta.getKind() == IJavaElementDelta.REMOVED) {
311          deletions.add(delta);
312        }
313        
314        // Recursively look for changes on children elements.
315        IJavaElementDelta[] children = delta.getAffectedChildren();
316        for (int i = 0; i < children.length; i++) {
317          traverse(children[i], additions, deletions);
318        }
319      }
