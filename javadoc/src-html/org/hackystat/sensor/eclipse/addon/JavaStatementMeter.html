001    package org.hackystat.sensor.eclipse.addon;
002    
003    import java.util.Iterator;
004    import java.util.List;
005    
006    import org.eclipse.jdt.core.dom.ASTVisitor;
007    import org.eclipse.jdt.core.dom.Annotation;
008    import org.eclipse.jdt.core.dom.Block;
009    import org.eclipse.jdt.core.dom.ExpressionStatement;
010    import org.eclipse.jdt.core.dom.IExtendedModifier;
011    import org.eclipse.jdt.core.dom.MethodDeclaration;
012    import org.eclipse.jdt.core.dom.MethodInvocation;
013    import org.eclipse.jdt.core.dom.Statement;
014    import org.eclipse.jdt.core.dom.TypeDeclaration;
015    
016    /**
017     * Implements a meter to measure several aspects such as number of methods, statements, test 
018     * methods, assertion statements if applicable. Number of test methods, assertion statements is
019     * reported to assist Test-Driven Development study.
020     * 
021     * @author Hongbing Kou
022     */
023    public class JavaStatementMeter extends ASTVisitor {
024      /** Name of this compilation unit. */
025      private String name; 
026      
027      /** Number of methods. */
028      private int numOfMethods = 0;
029      /** Number of invocations. */
030      private int numOfStatements = 0;
031      
032      /** Saves number of tests if there is any. */
033      private int numOfTestMethods = 0;
034      /** Saves number of test assertions in this class if there is any. */
035      private int numOfTestAssertions = 0;
036      
037      /**
038       * Visit class type declaration to get name of this object.
039       * 
040       * @param td TypeDeclaration such as class definition.
041       * @return True if continute to check child nodes. 
042       */
043      public boolean visit(TypeDeclaration td) {
044        if (td.getName() != null) {
045          this.name = td.getName().getIdentifier();
046        }
047        
048        return true;
049      }
050    
051      
052      /**
053       * Checks whether the given method is a JUnit 4 unit test.
054       * 
055       * @param md Method declaration.
056       * @return True if it is a unit test of JUnit 4. 
057       */
058      private boolean isJUnit4Test(MethodDeclaration md) {
059        List modifiers = md.modifiers();
060        for (Iterator i = modifiers.iterator(); i.hasNext();) {
061          IExtendedModifier modifer = (IExtendedModifier) i.next();
062          if (modifer.isAnnotation()) {
063            Annotation annotation = (Annotation) modifer;
064            if ("Test".equals(annotation.getTypeName().getFullyQualifiedName())) {
065              return true;
066            }
067          }
068        }
069        
070        return false;
071      }
072      
073      
074      /**
075       * Visits the method invocation to get number of assertion in this test method.
076       * 
077       * @param md Visit method declaration in source code.
078       * @return False because we don't look into child nodes. 
079       */
080      public boolean visit(MethodDeclaration md) {
081        if (md.getName() != null) {
082          this.numOfMethods++;
083          
084          if (md.getName().getIdentifier().startsWith("test")) {
085            this.numOfTestMethods++;
086          }
087          else if (isJUnit4Test(md)) {
088            this.numOfTestMethods++;
089          }
090          
091          // Check test method body to look for assertion statement. 
092          Block methodBody = md.getBody();
093          if (methodBody != null && methodBody.statements() != null) {
094            List stmts = methodBody.statements();
095            this.numOfStatements += stmts.size();
096            // Looks through all statements in this method body.
097            for (Iterator i = stmts.iterator(); i.hasNext();) {
098              Statement stmt = (Statement) i.next();
099              // MethodInvocation is one kind of expression statement.
100              if (stmt instanceof ExpressionStatement) {
101                ExpressionStatement estmt = (ExpressionStatement) stmt;
102                checkAssertions(estmt);
103              }
104            }
105          }
106        }
107            
108        //No need to visit child nodes anymore.
109        return false;
110      }
111    
112      /**
113       * Check expression statement for test assertions.
114       * 
115       * @param estmt Expression statements.
116       */
117      private void checkAssertions(ExpressionStatement estmt) {
118        if (estmt.getExpression() instanceof MethodInvocation) {
119          MethodInvocation mi = (MethodInvocation) estmt.getExpression();
120          // Increment number of test assertions.
121          if (mi.getName() != null && mi.getName().getIdentifier().startsWith("assert")) {
122            this.numOfTestAssertions++;
123          }
124        }
125      }
126      /**
127       * Gets number of methods.
128       * 
129       * @return Number of methods.
130       */
131      public int getNumOfMethods() {
132        return this.numOfMethods;
133      }
134      
135      /**
136       * Gets number of statements.
137       * 
138       * @return Number of statements.
139       */
140      public int getNumOfStatements() {
141        return this.numOfStatements;
142      }
143      
144      /**
145       * Gets number of tests.
146       * 
147       * @return Number of test.
148       */
149      public int getNumOfTestMethods() {
150        return this.numOfTestMethods;
151      }
152      
153      /**
154       * Gets number of test assertion statements.
155       * 
156       * @return Number of assertion statements
157       */
158      public int getNumOfTestAssertions() {
159        return this.numOfTestAssertions;
160      }
161      
162      /**
163       * Check whether this compilation unit has test. 
164       * 
165       * @return True if it has any test method or assertion.
166       */
167      public boolean hasTest() {
168        return this.numOfTestMethods > 0 || this.numOfTestAssertions > 0;   
169      }
170      
171      /**
172       * Returns meter values in string.
173       * 
174       * @return Metrics value string. 
175       */
176      public String toString() {
177        StringBuffer buf = new StringBuffer(200);
178        buf.append("*****  ").append(this.name).append("   *****\nMethods     : ")
179        .append(this.numOfMethods).append("\nStatements  : ").append(this.numOfStatements);
180            
181        // Appends test info if there is any.
182        if (this.hasTest()) {
183           buf.append("\nTests       : ").append(this.numOfTestMethods);
184           buf.append("\nAssertions  : ").append(this.numOfTestAssertions);
185        }
186        
187        return buf.toString();
188      }
189    }
