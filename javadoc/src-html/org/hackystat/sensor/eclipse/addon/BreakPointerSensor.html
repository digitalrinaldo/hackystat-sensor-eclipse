001    package org.hackystat.sensor.eclipse.addon;
002    
003    import java.net.URI;
004    import java.util.HashMap;
005    import java.util.Map;
006    
007    import org.eclipse.core.resources.IFile;
008    import org.eclipse.core.resources.IMarkerDelta;
009    import org.eclipse.core.runtime.CoreException;
010    import org.eclipse.debug.core.IBreakpointListener;
011    import org.eclipse.debug.core.model.IBreakpoint;
012    import org.eclipse.debug.core.model.ILineBreakpoint;
013    import org.eclipse.ui.IFileEditorInput;
014    import org.eclipse.ui.texteditor.ITextEditor;
015    import org.hackystat.sensor.eclipse.EclipseSensor;
016    import org.hackystat.sensor.eclipse.EclipseSensorPlugin;
017    
018    /**
019     * Listens to the break point changes to send activity data to Hackystat server.
020     * 
021     * @author Hongbing Kou
022     * @version $Id: BreakPointerSensor.java,v 1.1.1.1 2005/10/20 23:56:56 johnson Exp $
023     */
024    public class BreakPointerSensor implements IBreakpointListener {
025      /** Eclipse sensor. */
026      private EclipseSensor eclipseSensor;
027      
028      /**
029       * Creates a breakpoint listener instance.
030       * 
031       * @param eclipseSensor Eclipse sensor.
032       */
033      public BreakPointerSensor(EclipseSensor eclipseSensor) {
034        this.eclipseSensor = eclipseSensor;
035      }
036    
037      /**
038       * Listens to the break point added event.
039       * 
040       * @param breakpoint Break point.
041       */
042      public void breakpointAdded(IBreakpoint breakpoint) {
043        // Ignores breakpoints other than line break point
044        if (!(breakpoint instanceof ILineBreakpoint)) {
045          return;
046        }
047    
048        ITextEditor activeEditor = this.eclipseSensor.getActiveTextEditor();
049        if (activeEditor == null) {
050          return;
051        }
052        
053        IFileEditorInput fileEditorInput = (IFileEditorInput) activeEditor.getEditorInput();
054        IFile file = fileEditorInput.getFile();
055        try {
056          URI fileResource = file.getLocationURI();
057          String lineNumberString = String.valueOf(((ILineBreakpoint) breakpoint).getLineNumber());
058          
059          // Also generate DevEvent data for the debug event
060          Map<String, String> debugData = new HashMap<String, String>();
061          debugData.put("Subtype", "BreakPoint");
062          debugData.put("Set", "set");
063          debugData.put("Line", lineNumberString);
064          String message = constructMessage(fileResource, "Debug:BreakPoint", lineNumberString);
065          this.eclipseSensor.addDevEvent("Debug", fileResource, debugData, message);
066        }
067        catch (CoreException e) {
068          EclipseSensorPlugin.getInstance().log(e);
069        }
070      }
071    
072      /**
073       * Constructs message string to be displayed or logged.
074       * 
075       * @param fileResource File resource. 
076       * @param type Set/unset break point.
077       * @param lineno Line no.
078       * @return Message string.
079       */
080      private String constructMessage(URI fileResource, String type, String lineno) {
081        StringBuffer message = new StringBuffer();
082        message.append("Debug : ").append(this.eclipseSensor.extractFileName(fileResource));
083        message.append(" [").append(type).append(',').append(lineno).append(']');
084        
085        return message.toString();
086      }
087      
088      
089      /**
090       * Listens to break point changed event.
091       * 
092       * @param breakpoint Break point.
093       * @param delta Delta change.
094       */
095      public void breakpointChanged(IBreakpoint breakpoint, IMarkerDelta delta) {
096        // We don't need to do anything with changed breakpoint.
097      }
098    
099      /**
100       * Listens to break point removed event.
101       * 
102       * @param breakpoint Breakpoint.
103       * @param delta Marker changes.
104       */
105      public void breakpointRemoved(IBreakpoint breakpoint, IMarkerDelta delta) {
106        // Ignores breakpoints other than line break point
107        if (!(breakpoint instanceof ILineBreakpoint)) {
108          return;
109        }
110    
111        ITextEditor activeEditor = this.eclipseSensor.getActiveTextEditor();
112        IFileEditorInput fileEditorInput = (IFileEditorInput) activeEditor.getEditorInput();
113        IFile file = fileEditorInput.getFile();
114        try {
115          URI fileResource = file.getLocationURI();
116          String lineNumberString = String.valueOf(((ILineBreakpoint) breakpoint).getLineNumber());
117    
118          // Also generate DevEvent data for the debug event
119          Map<String, String> keyValueMap = new HashMap<String, String>();
120          keyValueMap.put("Subtype", "BreakPoint");
121          keyValueMap.put("Set", "unset");
122          keyValueMap.put("Line", lineNumberString);
123          String message = constructMessage(fileResource, "Debug:BreakPoint", keyValueMap.get("line"));
124          this.eclipseSensor.addDevEvent("Debug", fileResource, keyValueMap, message);
125        }
126        catch (CoreException e) {
127          // Exception can be thrown when there is no associated line number
128          // we do not need to worry about this in our sensor. 
129          EclipseSensorPlugin.getInstance().log(e);
130        }
131      }
132    }
