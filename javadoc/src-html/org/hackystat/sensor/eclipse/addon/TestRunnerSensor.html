018    /**
019     * Provides a JUnit listener implementing class to get JUnit result information. 
020     * A client must add this implementing class to the JUnitPlugin class in such a way that:
021     * 
022     * Note that this class extend the extension point defined in the JUnit plugin.

026    public class TestRunnerSensor implements ITestRunListener {
027      /** The EclipseSensor. */
028      private EclipseSensor sensor;
029      
030      /** Run time. */
031      private XMLGregorianCalendar runTime;

033      private int testCount;
034      /** Indice of the test in one test run. */
035      private int testIndice;

037      /** The starting time of each test case. */

062        init();
063      }

069        this.failureMessage = "";

085      /**
086       * A test run has started.
087       *
088       * @param testCount the number of tests that will be run.
089       *
090       * @see org.eclipse.jdt.junit.ITestRunListener#testRunStarted(int)
091       */
092      public void testRunStarted(int testCount) {
093        this.runTime = makeTimestamp();
094        this.testCount = testCount;
095        this.testIndice = 1;
096      }
097      
098      /**
099       * Converts the specified time in milliseconds into a
100       * javax.xml.datatype.XMLGregorianCalendar.
101       * @return A new instance of a javax.xml.datatype.XmlGregorianCalendar
102       */
103      private XMLGregorianCalendar makeTimestamp() {
104        DatatypeFactory factory = null;
105        try {
106          factory = DatatypeFactory.newInstance();
107          GregorianCalendar calendar = new GregorianCalendar();
108          calendar.setTimeInMillis(new Date().getTime());
109          return factory.newXMLGregorianCalendar(calendar);
110        }
111        catch (DatatypeConfigurationException e) {
112          throw new RuntimeException("Can not get date of XMLGregorianCalendar", e);
113        }
114      }

144      /**
145       * Gets the fully qualified package class name from the testName string. For example, returns
146       * "org.hackystat.sensor.eclipse.junit.EclipseJUnitListener".
147       *
148       * @param testName The test name string to be passed.
149       *
150       * @return The fully qualified package class name.
151       */
152      private String getTestNameOf(String testName) {
153        int startIndex = testName.indexOf('(');
154        int endIndex = testName.indexOf(')');
155    
156        // There is no test case
157        if (startIndex == -1 || endIndex == -1) {
158          return "";
159        }
160        
161        return testName.substring(startIndex + 1, endIndex);
162      }
163    
164      /**
165       * Gets the test case name (method name) from the testName string. For example, returns
166       * "testMethod".
167       *
168       * @param test The test name string to be passed.
169       *
170       * @return The test case name (method name).
171       */
172      private String getTestCaseOf(String test) {
173        int endIndex = test.indexOf('(');
174        
175        // There is no test but test case name.
176        if (endIndex == -1) {
177          return test;
178        }
179        
180        return test.substring(0, endIndex);
181      }

183      /**
184       * An individual test has started. Supports the method for the 2.1 stream from RC2.
185       *
186       * @param testId a unique Id identifying the test
187       * @param test the name of the test that started
188       *
189       * @see org.eclipse.jdt.junit.ITestRunListener#testStarted(java.lang.String, java.lang.String)
190       * @since 2.1 RC2
191       **/
192      public void testStarted(String testId, String test) {
193        this.startTime = new Date();
194        init();
195      }
196    
197      /**
198       * An individual test has ended. Supports the method for the 2.1 stream from RC2.
199       *
200       * @param testId a unique Id identifying the test
201       * @param test the name of the test that ended
202       *
203       * @see org.eclipse.jdt.junit.ITestRunListener#testEnded(java.lang.String, java.lang.String)
204       * @since 2.1 RC2
205       **/
206      public void testEnded(String testId, String test) {
207        Date endTime = new Date();
208        String testName = getTestNameOf(test);
209        String testCase = getTestCaseOf(test);
210        
211        String name = testName + "." + testCase;
212        // Incase there is compilation error so that there is not test name
213        if ("".equals(testName)) {
214          name = testCase;
215        }
216        
217        // Run test
218        Map<String, String> keyValueMap = new HashMap<String, String>();
219        keyValueMap.put("Runtime", String.valueOf(this.runTime.toString()));
220    
221        // Elapsed time
222        long elapsedTime = endTime.getTime() - this.startTime.getTime();
223        elapsedTime = (elapsedTime >= 0) ? elapsedTime : 0;
224        keyValueMap.put("ElapsedTime", String.valueOf(elapsedTime));
225        
226        String result = "pass";
227        if (this.failureMessage.length() > 0) {
228          result = "fail";
229          keyValueMap.put("FailureString", this.failureMessage);      
230        }
231        else if (this.errorMessage.length() > 0) {
232          result = "fail";
233          keyValueMap.put("ErrorString" , this.errorMessage);      
234        }
235        keyValueMap.put("TestCase", testCase); // Test case
236        keyValueMap.put("TestName", testName); // Test name
237        
238        keyValueMap.put("TestCount", String.valueOf(this.testCount));
239        keyValueMap.put("TestIndice", String.valueOf(this.testIndice));
240    
241        URI testFileResource = this.sensor.getObjectFile(testName);
242        if (testFileResource == null) {
243          testFileResource = this.sensor.getActiveFile();
244        }
245        
246        // Increment indice number by one. 
247        this.testIndice++;
248        // Process UnitTest
249        this.sensor.addDevEvent("Test", testFileResource, keyValueMap, name + " : " + result);
250      }

252      /**
253       * An individual test has failed with a stack trace.
254       * Supports the method for the 2.1 stream from RC2.
255       *
256       * @param testId a unique Id identifying the test
257       * @param test the name of the test that failed
258       * @param status the outcome of the test; one of
259       * {@link #STATUS_ERROR STATUS_ERROR} or
260       * {@link #STATUS_FAILURE STATUS_FAILURE}
261       * @param trace the stack trace
262       *
263       * @see org.eclipse.jdt.junit.ITestRunListener#testFailed(int, java.lang.String,
264       * java.lang.String, java.lang.String)
265       * @since 2.1 RC2
266       */
267      public void testFailed(int status, String testId, String test, String trace) {
268        if (status == ITestRunListener.STATUS_FAILURE) {
269          this.failureMessage = trace;
270        }
271        else if (status == ITestRunListener.STATUS_ERROR) {
272          this.errorMessage = trace;
273        }
274      }

277       * An individual test has been rerun.
278       * Supports the method for the 2.1 stream from RC2.
279       *
280       * @param testId a unique Id identifying the test
281       * @param testClass the name of the test class that was rerun
282       * @param test the name of the test that was rerun
283       * @param status the outcome of the test that was rerun; one of
284       * {@link #STATUS_OK STATUS_OK}, {@link #STATUS_ERROR STATUS_ERROR},
285       * or {@link #STATUS_FAILURE STATUS_FAILURE}
286       * @param trace the stack trace in the case of abnormal termination,
287       * or the empty string if none
288       *
289       * @see org.eclipse.jdt.junit.ITestRunListener#testReran(java.lang.String, java.lang.String,
290       *  java.lang.String, int, java.lang.String)
291       * @since 2.1 RC2
292       */
293      public void testReran(String testId, String testClass,
294                            String test, int status, String trace) {
295        testReran(testClass, test, status, trace);
296      }
