001    package org.hackystat.sensor.eclipse.addon;
002    
003    import java.net.URI;
004    import java.util.HashMap;
005    import java.util.Map;
006    
007    import org.eclipse.core.resources.IFile;
008    import org.eclipse.debug.core.DebugEvent;
009    import org.eclipse.debug.core.IDebugEventSetListener;
010    import org.eclipse.debug.core.model.IDebugTarget;
011    import org.eclipse.ui.IFileEditorInput;
012    import org.eclipse.ui.texteditor.ITextEditor;
013    import org.hackystat.sensor.eclipse.EclipseSensor;
014    
015    /**
016     * Implements debug listener to intercept debug activity.
017     *  
018     * @author Hongbing Kou
019     * @version $Id: DebugSensor.java,v 1.1.1.1 2005/10/20 23:56:56 johnson Exp $
020     */
021    public class DebugSensor implements IDebugEventSetListener {
022      /** The file being debugged. */
023      private URI debuggedFile;
024      /** Eclipse sensor. */
025      private EclipseSensor eclipseSensor;
026      
027      /**
028       * Instantiates an Eclipse sensor.
029       * 
030       * @param sensor Eclipse sensor.
031       */
032      public DebugSensor(EclipseSensor sensor) {
033        this.eclipseSensor = sensor;
034      }
035      
036      /**
037       * Handle debug event to get execution information of the program being developed.
038       * 
039       * @param events Debug events being fired.
040       */
041      public void handleDebugEvents(DebugEvent[] events) {
042        if (events != null && events.length > 0) {
043          for (int i = 0; i < events.length; i++) {
044            DebugEvent event = events[i];
045            String debugActivity = translateEventKind(event);
046            if (debugActivity != null) {
047              StringBuffer displayMessage = new StringBuffer();
048              
049              Map<String, String> keyValueMap = new HashMap<String, String>();
050              keyValueMap.put("Subtype", debugActivity);
051              
052              displayMessage.append("Debug : ").append(
053                this.eclipseSensor.extractFileName(this.debuggedFile));
054              displayMessage.append(" [").append(debugActivity).append(']');
055              
056              this.eclipseSensor.addDevEvent("Debug", this.debuggedFile, 
057                keyValueMap, displayMessage.toString());
058            }
059          }
060        }
061      }
062      
063      /**
064       * Translates debug event from event kind and detail to meaningful 
065       * string representation.
066       * 
067       * @param event Debug event.
068       * @return String representation.
069       */
070      private String translateEventKind(DebugEvent event) {
071        String debugType = null;
072        if (event.getKind() == DebugEvent.CREATE) {
073          if (event.getSource() instanceof IDebugTarget) {
074            debugType = "Start";
075            
076            // Update the file being edited
077            ITextEditor activeEditor = this.eclipseSensor.getActiveTextEditor();
078            IFileEditorInput fileEditorInput = (IFileEditorInput) activeEditor.getEditorInput();
079            IFile file = fileEditorInput.getFile();
080            this.debuggedFile = file.getLocationURI();
081          }
082        }
083        else if (event.getKind() == DebugEvent.TERMINATE) {
084          if (event.getSource() instanceof IDebugTarget) {
085            debugType = "Terminate";
086          }
087        }
088        else if (event.getKind() == DebugEvent.SUSPEND || 
089                 event.getKind() == DebugEvent.RESUME) {
090          switch (event.getDetail()) {
091            case DebugEvent.BREAKPOINT:
092              debugType = "Breakpoint";
093              break;
094            case DebugEvent.STEP_INTO:
095              debugType = "Step into";
096              break;
097            case DebugEvent.STEP_OVER:
098              debugType = "Step over";
099              break;
100            default:
101              debugType = "Unknown";
102          }
103        }
104        
105        return debugType;      
106      }  
107    }
