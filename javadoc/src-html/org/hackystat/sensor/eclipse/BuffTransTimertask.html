001    package org.hackystat.sensor.eclipse;
002    
003    import java.util.TimerTask;
004    
005    import org.hackystat.sensorshell.SensorShellException;
006    
007    /**
008     * Implements Eclipse sensor timer task that can be executed by a timer. The timer task checks
009     * buffer transition, which represents the event where a developer moves from one buffer
010     * (containing a file) to another buffer (containing a different file).
011     *
012     * @author Hongbing Kou
013     */
014    public class BuffTransTimertask extends TimerTask {
015      /**
016       * Processes the state change activity and computes file metrics in a time based interval.
017       */
018      public void run() {
019        try {
020          EclipseSensor sensor = EclipseSensor.getInstance();
021          // process buffer transactions.
022          sensor.processBuffTrans();
023        }
024        catch (SensorShellException e) {
025          e.printStackTrace();
026        }
027      }
028    }
