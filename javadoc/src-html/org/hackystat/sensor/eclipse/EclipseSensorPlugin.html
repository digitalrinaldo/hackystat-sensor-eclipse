104      /**
105       * Implements an inner thread class to handle sensor update.  
106       */
107      private class SensorUpdateThread extends Thread {
108        /** Resource bundle. */
109        private Bundle bundle;
110        /** Update site. */
111        private String updateSite; 
112        
113        /**
114         * Hackystat sensor update thread.
115         * 
116         * @param bundle Plugin bundle.
117         * @param updateSite Update site.
118         */
119        SensorUpdateThread(Bundle bundle, String updateSite) {
120          this.setName("SensorUpdateThread");
121          this.updateSite = updateSite;
122        }
123        
124        /**
125         * Thread execution function.
126         */
127        public void run() {
128          String title = EclipseSensorI18n.getString("VersionCheck.messageDialogTitle");
129          String first = EclipseSensorI18n.getString("VersionCheck.messageDialogMessageFirst");
130          String betweenKey = "VersionCheck.messageDialogMessageBetween";
131          String between = EclipseSensorI18n.getString(betweenKey);
132          String last = EclipseSensorI18n.getString("VersionCheck.messageDialogMessageLast");
133          String messages[] = {first, between, last};
134          
135          VersionCheck versionCheck = new VersionCheck(this.bundle); 
136          versionCheck.processUpdateDialog(updateSite, title, messages);
137        }
138      }
139      
140      /**
141       * Gets the path to the sensorshell.jar.
142       * 
143       * @return the path to the sensorshell.jar.
144       */
145      public String getSensorShellPath() {
146        URL pluginUrl = super.getBundle().getEntry("/");
147        try {      
148          return FileLocator.toFileURL(new URL(pluginUrl, "sensorshell.jar")).getFile();
149        }
150        catch (Exception e) {
151          return null;
152        }
153      }
154

167      /**
168       * Logs out the exception or error message for Eclispe sensor plug-in.
169       * 
170       * @param message Error message.
171       * @param e Exception. 
172       */
173      public void log(String message, Exception e) {
174        String pluginName = super.getBundle().getSymbolicName();
175        IStatus status = new Status(IStatus.ERROR, pluginName, 0, message + " " + e.getMessage(), e);
176        
177        plugin.getLog().log(status);
178      }
