127      /**
128       * Provides instantiation of SensorProperties, which has information in the sensor.properties
129       * file, and executes <code>doCommand</code> to activate sensor. Note that the Eclipse instance
130       * is lazily instantiated when static <code>getInstance()</code> was called.
131       * 
132       * @throws SensorShellException If error in getting sensorshell.
133       */
134      private EclipseSensor() throws SensorShellException {
135        this.timer = new Timer();
136        this.stateChangeTimerTask = new StateChangeTimerTask();
137        this.buffTransTimerTask = new BuffTransTimertask();
138        
139        // Load sensor's setting. 
140        this.hotDeploySensorShell();
141        
142        // Adds this EclipseSensorPlugin instance to IResourceChangeListener
143        // so that project event and file save event is notified.
144        IWorkspace workspace = ResourcesPlugin.getWorkspace();
145    
146        // Adds IResourceChangeListener-implemented ResourceChangeAdapter before checking isEnabled
147        // for piemontese sensor because this listener is used in piemontese sensor even though
148        // the main sensor.properties does not exist in the <hackystat_home>.
149        workspace.addResourceChangeListener(new ResourceChangeAdapter(),
150            IResourceChangeEvent.POST_CHANGE);
151    
152        // Adds element changed listener to get the corresponding change of refactoring.
153        JavaCore.addElementChangedListener(new JavaStructureChangeDetector(this));
154        
155        initialize();
156      }

212      /**
213       * Initializes sensor and JUnitListener instance if the sensor is enabled. Note that JUnit
214       * listener instance is added only when the instance is not instantiated.
215       */
216      private void initialize() {
217        // Sets state change time schedule.
218        if (this.stateChangeTimerTask.scheduledExecutionTime() == 0) {
219          this.timer.schedule(this.stateChangeTimerTask, this.timerStateChangeInterval * 1000,
220                this.timerStateChangeInterval * 1000);
221        }
222    
223        // Sets buffer transition time schedule.
224        if (this.buffTransTimerTask.scheduledExecutionTime() == 0) {
225          this.timer.schedule(this.buffTransTimerTask, this.timeBuffTransInterval * 1000,
226                this.timeBuffTransInterval * 1000);
227         }
228    
229        registerListeners();
230      }

232      /**
233       * Provide the initialization of the listeners additions. The Window, Part, and Document Listener
234       * are added. Note that sensor shell should be instantiated before this method is called because
235       * <code>processActivity()</code> method uses sensor shell instance.
236       */
237      private void registerListeners() {
238        IWorkbench workbench = EclipseSensorPlugin.getInstance().getWorkbench();
239    
240        // :RESOLVED: JULY 1, 2003
241        // Supports the multiple window for sensor collection.
242        IWorkbenchWindow[] activeWindows = workbench.getWorkbenchWindows();
243    
244        // Check if window listener is not added yet. Otherwise multi instances are notified.
245        if (this.windowListener == null) {
246          this.windowListener = new WindowListenerAdapter();
247          workbench.addWindowListener(new WindowListenerAdapter());
248        }
249    
250        for (int i = 0; i < activeWindows.length; i++) {
251          IWorkbenchPage activePage = activeWindows[i].getActivePage();
252          activePage.addPartListener(new PartListenerAdapter());
253          IEditorPart activeEditorPart = activePage.getActiveEditor();
254    
255          // Adds this EclipseSensorPlugin instance to IDocumentListener
256          // only when activeEditorPart is the instance of ITextEditor
257          // so that null case is also ignored.
258          if (activeEditorPart instanceof ITextEditor) {
259            // Sets activeTextEditor. Otherwise a first activated file would not be recorded.
260            this.activeTextEditor = (ITextEditor) activeEditorPart;
261            // Gets opened file since the initial opened file is not notified from IPartListener.
262            URI fileResource = EclipseSensor.this.getFileResource(this.activeTextEditor);
263    
264            Map<String, String> keyValueMap = new HashMap<String, String>();
265            keyValueMap.put(EclipseSensorConstants.SUBTYPE, "Open");
266            keyValueMap.put(EclipseSensorConstants.UNIT_TYPE, EclipseSensorConstants.FILE);
267            keyValueMap.put(EclipseSensorConstants.UNIT_NAME, 
268                EclipseSensor.this.extractFileName(fileResource));
269            this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, fileResource, keyValueMap, 
270                "Opened " + fileResource.toString());
271    
272            IDocumentProvider provider = this.activeTextEditor.getDocumentProvider();
273            IDocument document = provider.getDocument(activeEditorPart.getEditorInput());
274    
275            // Initially sets active buffer and threshold buffer.
276            // Otherwise a first activated buffer would not be recorded.
277            this.activeBufferSize = document.getLength();
278            this.thresholdBufferSize = document.getLength();
279            document.addDocumentListener(new DocumentListenerAdapter());
280          }
281        }
282    
283        // Handles breakpoint set/unset event.
284        IBreakpointManager bpManager = DebugPlugin.getDefault().getBreakpointManager();
285        bpManager.addBreakpointListener(new BreakPointerSensor(this));
286    
287        // Listens to debug event.
288        DebugPlugin.getDefault().addDebugEventListener(new DebugSensor(this));
289    
290        // Creates instance to handle build error.
291        this.buildErrorSensor = new BuildErrorSensor(this);
292      }

329        String fileStirng = fileResource.toString();

345      /**
346       * Process the state change activity whose element consists of the (absolute) file name and its
347       * buffer size (or file size).
348       */
349      public void processStateChangeActivity() {
350        if (this.activeTextEditor == null) {
351          return;
352        }
353        
354        URI fileResource = this.getFileResource(this.activeTextEditor);
355        if (fileResource != null) {
356          int activeBufferSize = this.activeBufferSize;
357    
358          // Will not send out data if there is no state change at all.
359          if (fileResource.equals(this.latestStateChangeFile)
360              && this.latestStateChangeFileSize == activeBufferSize) {
361            return;
362          }
363    
364          // Makes up state change data
365          //StringBuffer statechangeData = new StringBuffer();
366          //statechangeData.append(activeFileName);
367    
368          // Calculate test methods and assertions if it is a java file.
369          IFileEditorInput fileEditorInput = (IFileEditorInput) this.activeTextEditor.getEditorInput();
370          IFile file = fileEditorInput.getFile();
371          if (file.exists()) {
372            Map<String, String> statechangeMetricsMap = computeFileMetrics(file);  
373    
374            // Status message for display
375            StringBuffer msgBuf = new StringBuffer();
376            msgBuf.append("statechange : ").append(file.getName());
377            msgBuf.append(" [").append(statechangeMetricsMap.get(PROP_CURRENT_SIZE));
378            
379            String className = statechangeMetricsMap.get(PROP_CLASS_NAME);
380            if (className != null) {
381              msgBuf.append(", ").append(className);
382            }
383            
384            String currentMethods = statechangeMetricsMap.get(PROP_CURRENT_METHODS); 
385            if (currentMethods != null) {
386              msgBuf.append(", methods=").append(currentMethods);
387            }
388            
389            String currentStatements = statechangeMetricsMap.get(PROP_CURRENT_STATEMENTS);
390            if (currentStatements != null) {
391              msgBuf.append(", stms=").append(currentStatements);
392            }
393            msgBuf.append(']');
394    
395            this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, fileResource, statechangeMetricsMap,
396                msgBuf.toString());
397          }
398          
399          this.latestStateChangeFile = fileResource;
400          this.latestStateChangeFileSize = activeBufferSize;
401        }
402      }

404      /**
405       * Computes file metrics.
406       * 
407       * @param file File.
408       * @return Map that contains key-value pairs if object metrics.
409       */
410      private Map<String, String> computeFileMetrics(IFile file) {
411        // Adds statechange DevEvent.
412        Map<String, String> statechangeKeyValueMap = new HashMap<String, String>();
413        statechangeKeyValueMap.put(EclipseSensorConstants.SUBTYPE, "StateChange");
414        // Process file size
415        String sizeString = String.valueOf(activeBufferSize);
416        statechangeKeyValueMap.put(PROP_CURRENT_SIZE, sizeString);
417        
418        if (file.getName().endsWith(EclipseSensorConstants.JAVA_EXT)) {
419          // Fully qualified class path
420          String className = this.getFullyQualifedClassName(file);
421          statechangeKeyValueMap.put(PROP_CLASS_NAME, className);
422    
423          // Measure java file.
424          JavaStatementMeter testCounter = measureJavaFile(file);
425          
426          this.class2FileMap.put(className, file.getLocationURI());
427          String methodCountString = String.valueOf(testCounter.getNumOfMethods());
428          statechangeKeyValueMap.put(PROP_CURRENT_METHODS, methodCountString);
429    
430          String statementCountString = String.valueOf(testCounter.getNumOfStatements());
431          statechangeKeyValueMap.put(PROP_CURRENT_STATEMENTS, 
432              statementCountString);
433    
434          // Number of test method and assertion statements.
435          if (testCounter.hasTest()) {
436            String testMethodCount = String.valueOf(testCounter.getNumOfTestMethods());
437            statechangeKeyValueMap.put(PROP_CURRENT_TEST_METHODS, 
438                testMethodCount);
439            
440            String testAssertionCount = String.valueOf(testCounter.getNumOfTestAssertions()); 
441            statechangeKeyValueMap.put(PROP_CURRENT_TEST_ASSERTIONS, testAssertionCount);
442          }
443        }
444        
445        return statechangeKeyValueMap;
446      }

448      /**
449       * Calculates java file unit test information.
450       *
451       * @param file IFile instance to a java file.
452       *
453       * @return UnitTestCounter instance to this java file.
454       */
455      private JavaStatementMeter measureJavaFile(IFile file) {
456        // Compute number of tests and assertions to this file.
457        ICompilationUnit cu = (ICompilationUnit) JavaCore.create(file);
458        ASTParser parser = ASTParser.newParser(AST.JLS3);
459        parser.setSource(cu);
460        parser.setResolveBindings(true);
461    
462        ASTNode root = parser.createAST(null);
463        JavaStatementMeter counter = new JavaStatementMeter();
464        root.accept(counter);
465    
466        return counter;
467      }

469      /**
470       * Process the buffer transition to check if the current buffer is visiting a file and if that
471       * file is different from the file visited by the buffer during the last wakeup. Its element
472       * consists of the the (absolute) file name (or last-time-visited file name) from which an user is
473       * visiting, the (absolute) file name (or current-visiting file name) to which the user is
474       * visiting, and the modification status of the last-time-visited file.
475       */
476      public void processBuffTrans() {
477        // check if BufferTran property is enable
478        if (this.activeTextEditor == null || (this.previousTextEditor == null)) {
479          return;
480        }
481        URI toFile = this.getFileResource(this.activeTextEditor);
482        URI fromFile = this.getFileResource(this.previousTextEditor);
483        if (fromFile != null && toFile != null && !toFile.equals(fromFile)) {
484          String buffTrans = fromFile.toString() + "->" + toFile.toString();
485          // :RESOVED: 5/21/04 ISSUE:HACK109
486          if (!latestBuffTrans.equals(buffTrans)) {
487            HashMap<String, String> buffTranKeyValuePairs = new HashMap<String, String>();
488            
489            buffTranKeyValuePairs.put(EclipseSensorConstants.SUBTYPE, "BufferTransition");
490            buffTranKeyValuePairs.put("From-Buff-Name", fromFile.toString());
491            buffTranKeyValuePairs.put("To-Buff-Name", toFile.toString());
492            buffTranKeyValuePairs.put("Modified", String.valueOf(this.isModifiedFromFile));
493            
494            String message = "BuffTrans : " + this.extractFileName(fromFile) + " --> " +
495                              this.extractFileName(toFile);
496    
497            this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, toFile, buffTranKeyValuePairs, message);
498            latestBuffTrans = buffTrans;
499          }
500        }
501      }

503      /**
504       * Gets the fully qualified class name for an active file. For example, its value is foo.bar.Baz.
505       *
506       * @param file Get fully qualified class file.
507       * @return The fully qualified class name. For example,foo.bar.Baz.
508       */
509      private String getFullyQualifedClassName(IFile file) {
510        String fullClassName = "";
511        if (file.exists() && file.getName().endsWith(EclipseSensorConstants.JAVA_EXT)) {
512          ICompilationUnit compilationUnit = (ICompilationUnit) JavaCore.create(file);
513          String className = compilationUnit.getElementName();
514          if (className.endsWith(EclipseSensorConstants.JAVA_EXT)) {
515            className = className.substring(0, className.length() - 5);
516          }
517          
518          try {
519            IPackageDeclaration[] packageDeclarations = compilationUnit.getPackageDeclarations();
520            // Should only have one package declaration
521            if (packageDeclarations == null || packageDeclarations.length == 0) {
522              fullClassName = className;
523            }
524            else {
525              fullClassName = packageDeclarations[0].getElementName() + '.' +
526                              className;
527            }
528          }
529          catch (JavaModelException e) {
530            // This exception will be thrown if user is working on a Java but did not open
531            // it with "Java Perspective". Thus, the Java Model does not exist to parse 
532            // Java files. So we only log out exception while Eclipse's Java Perspective 
533            // exits.
534            if (!e.isDoesNotExist()) {
535               EclipseSensorPlugin.getDefault().log(file.getName(), e);
536            }
537          }
538        }
539        
540        return fullClassName;
541      }

543      /**
544       * Gets the URI of the associated file. It will return file URI in format such as 
545       * file:/D:/cvs/foobarproject/src/foo/bar/Bar.java. When a developers is not editing
546       * an active file, this method will return a dummy file resource.
547       *
548       * @param textEditor A ITextEditor instance form which the file name is retrieved.
549       * @return The fully qualified file name. For example, C:\cvs\foobarproject\src\foo\bar\Bar.java.
550       */
551      private URI getFileResource(ITextEditor textEditor) {
552        if (textEditor != null) {
553          IEditorInput editorInput = textEditor.getEditorInput();
554          if (editorInput instanceof IFileEditorInput) {
555            IFileEditorInput input = (IFileEditorInput) editorInput;
556            IFile file = input.getFile();
557            if (file != null) {
558              URI fileResource = file.getLocationURI();
559              return fileResource;
560            }
561          }
562        }
563    
564        // Return a dummy unknown resource if can not resolve the file being edited. 
565        return URI.create("file:///Unknown");
566      }

597      /**
598       * Provides the IWindowListener-implemented class to catch the "Browser activated", "Browser
599       * closing" event. This inner class is designed to be used by the outer EclipseSensor class.
600       *
601       * @author Takuya Yamashita
602       * @version $Id: EclipseSensor.java,v 1.1.1.1 2005/10/20 23:56:56 johnson Exp $
603       */
604      private class WindowListenerAdapter implements IWindowListener {
605        /**
606         * Provides manipulation of browser open status due to implement <code>IWindowListener</code>.
607         * This method must not be called by client because it is called by platform. Do nothing for
608         * Eclipse sensor so far.
609         *
610         * @param window An IWorkbenchWindow instance to be triggered when a window is activated.
611         */
612        public void windowActivated(IWorkbenchWindow window) {
613          IEditorPart activeEditorPart = window.getActivePage().getActiveEditor();
614          if (activeEditorPart instanceof ITextEditor) {
615            EclipseSensor.this.activeTextEditor = (ITextEditor) activeEditorPart;
616            ITextEditor editor = EclipseSensor.this.activeTextEditor;

738          if (part instanceof ITextEditor) {
739            URI fileResource = EclipseSensor.this.getFileResource((ITextEditor) part);
740            Map<String, String> keyValueMap = new HashMap<String, String>();
741            keyValueMap.put(EclipseSensorConstants.SUBTYPE, "Close");
742            if (fileResource != null 
743                && fileResource.toString().endsWith(EclipseSensorConstants.JAVA_EXT)) {
744              keyValueMap.put("Language", "java");
745            }
746            keyValueMap.put(EclipseSensorConstants.UNIT_TYPE, EclipseSensorConstants.FILE);
747            keyValueMap.put(EclipseSensorConstants.UNIT_NAME, EclipseSensor.this.extractFileName(fileResource));
748            EclipseSensor.this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, 
749                fileResource, keyValueMap, fileResource.toString());
750            IEditorPart activeEditorPart = part.getSite().getPage().getActiveEditor();
751            if (activeEditorPart == null) {
752              EclipseSensor.this.activeTextEditor = null;
753            }
754          }

757        /**
758         * Provides manipulation of browser part deactivation status due to implement
759         * <code>IPartListener</code>. This method must not be called by client because it is called
760         * by platform. Sets active text editor to be null when the text editor part is deactivated.
761         *
762         * @param part An IWorkbenchPart instance to be triggered when a part is deactivated.
763         */
764        public void partDeactivated(IWorkbenchPart part) {
765          if (part instanceof ITextEditor && !part.equals(EclipseSensor.this.deactivatedTextEditor)) {
766            EclipseSensor.this.deactivatedTextEditor = (ITextEditor) part;
767            if (EclipseSensor.this.isActivatedWindow) {
768              IEditorPart activeEditorPart = part.getSite().getPage().getActiveEditor();
769    
770              // Sets activeTextEdtior to be null only when there is no more active editor.
771              // Otherwise the case that the non text editor part is active causes the activeTextEditor
772              // to be null so that sensor is not collected after that.
773              if (activeEditorPart == null) {
774                EclipseSensor.this.activeTextEditor = null;
775              }
776    
777              // BuffTrans to get the toFrom buffer size.
778              ITextEditor editor = (ITextEditor) part;
779              IDocumentProvider provider = editor.getDocumentProvider();
780    
781              // provider could be null if the text editor is closed before this method is called.
782              EclipseSensor.this.isModifiedFromFile = false;
783              EclipseSensor.this.previousTextEditor = null;
784    
785              if (provider != null) {
786                EclipseSensor.this.previousTextEditor = editor;
787                int fromFileBufferSize = provider.getDocument(editor.getEditorInput()).getLength();
788    
789                // Check if a threshold buffer is either dirty or
790                // not the same as the current from file buffer size;
791                EclipseSensor.this.isModifiedFromFile = (editor.isDirty()
792                    || (EclipseSensor.this.thresholdBufferSize != fromFileBufferSize));
793              }
794            }
795            else {
796              EclipseSensor.this.isActivatedWindow = true;
797            }
798          }
799        }

856      /**
857       * Provides "Open Project, "Close Project", and "Save File" events. Note that this implementing
858       * class uses Visitor pattern so that key point to gather these event information is inside the
859       * visitor method which is implemented from <code>IResourceDeltaVisitor</code> class.
860       *
861       * @author Takuya Yamashita
862       * @version $Id: EclipseSensor.java,v 1.1.1.1 2005/10/20 23:56:56 johnson Exp $
863       */
864      private class ResourceChangeAdapter implements IResourceChangeListener, IResourceDeltaVisitor {
865        /**
866         * Provides manipulation of IResourceChangeEvent instance due to implement
867         * <code>IResourceChangeListener</code>. This method must not be called by client because it
868         * is called by platform when resources are changed.
869         *
870         * @param event A resource change event to describe changes to resources.
871         */
872        public void resourceChanged(IResourceChangeEvent event) {
873          if (((event.getType() & IResourceChangeEvent.POST_CHANGE) != 0)) {
874            //          ||
875            //          ((event.getType() & IResourceChangeEvent.POST_AUTO_BUILD) != 0)) {
876            try {
877              IResourceDelta rootDelta = event.getDelta();
878    
879              // Accepts the class instance to let the instance be able to visit resource delta.
880              rootDelta.accept(this);
881            }
882            catch (CoreException e) {
883              e.printStackTrace();
884            }
885          }
886        }
887    
888        /**
889         * Provides visitor pattern due to implement <code>IResourceDeltaVisitor</code>. This method
890         * must not be called by client because it is called by EclipseSensorPlugin instance. Note that
891         * <code>true</code> is returned if the parameter of IResourceDelta instance has children.
892         * <code>false</code> is returned when either Project is opened, closed, or file is saved
893         * because no more traverse of children of the IResourceDelta instance is needed.
894         *
895         * @param delta IResourceDelta instance to contains delta resource.
896         * @return true if the resource delta's children should be visited; false if they should be
897         *         skipped.
898         * @throws CoreException if the visit fails for some reason.
899         */
900        public boolean visit(IResourceDelta delta) throws CoreException {
901          IResource resource = delta.getResource();
902          int flag = delta.getFlags();
903          int kind = delta.getKind();
904    
905          // If there is compilation problem with the current java file then send out the activity data.
906          if ((flag & IResourceDelta.MARKERS) != 0 && EclipseSensor.this.buildErrorSensor != null) {
907            EclipseSensor.this.buildErrorSensor.findBuildProblem(delta);
908          }
909    
910          // :RESOLVED: 26 May 2003
911          // Note that the 147456 enumeration type is not listed in the IResourceDelta static filed.
912          // However, its number is generated when Project is either opened or closed so that
913          // it is checked in the logical condition.
914          if (resource instanceof IProject && ((flag == IResourceDelta.OPEN) || (flag == 147456))) {
915            IProject project = resource.getProject();
916            String projectName = project.getName();
917            URI projectResoruce = project.getFile(".project").getLocationURI();
918    
919            Map<String, String> keyValueMap = new HashMap<String, String>();
920            keyValueMap.put(EclipseSensorConstants.UNIT_TYPE, "project");
921            keyValueMap.put(EclipseSensorConstants.UNIT_NAME, projectName);
922    
923            if (((IProject) resource).isOpen()) {
924              keyValueMap.put(EclipseSensorConstants.SUBTYPE, "Open");
925              EclipseSensor.this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, projectResoruce, 
926                keyValueMap, projectResoruce.toString());          
927            }
928            else {
929              keyValueMap.put(EclipseSensorConstants.SUBTYPE, "Close");
930              EclipseSensor.this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, projectResoruce, 
931                keyValueMap, projectResoruce.toString());          
932            }
933            return false;
934          }
935          if ((kind == IResourceDelta.CHANGED) && (flag == IResourceDelta.CONTENT)
936              && resource instanceof IFile) {
937            if (resource.getLocation().toString().endsWith(EclipseSensorConstants.JAVA_EXT)) {
938              IFile file = (IFile) resource;
939    
940              Map<String, String> keyValueMap = new HashMap<String, String>();
941              
942              keyValueMap.put("Language", "java");
943              keyValueMap.put(EclipseSensorConstants.UNIT_TYPE, EclipseSensorConstants.FILE);
944              
945              // Fully qualified class path
946              String className = EclipseSensor.this.getFullyQualifedClassName(file);
947              keyValueMap.put("Class-Name", className);
948    
949              // Size of the file in buffer
950              String bufferSize = String.valueOf(activeBufferSize);
951              keyValueMap.put("Current-Size", bufferSize);
952    
953              // Measure java file.
954              JavaStatementMeter testCounter = measureJavaFile(file);
955              testCounter = measureJavaFile(file);
956              String methodCount = String.valueOf(testCounter.getNumOfMethods());
957              keyValueMap.put("Current-Methods", methodCount);
958              
959              String statementCount = String.valueOf(testCounter.getNumOfStatements());
960              keyValueMap.put("Current-Statements", statementCount);
961    
962              // Number of test method and assertion statements.
963              if (testCounter.hasTest()) {
964                String testMethodCount = String.valueOf(testCounter.getNumOfTestMethods());
965                keyValueMap.put("Current-Test-Methods", testMethodCount);
966                
967                String testAssertionCount = String.valueOf(testCounter.getNumOfTestAssertions());
968                keyValueMap.put("Current-Test-Assertions", testAssertionCount);
969              }
970    
971              //EclipseSensor.this.eclipseSensorShell.doCommand("Activity", activityData);
972              //Construct message to display on Eclipse status bar.
973              URI fileResource = file.getLocationURI();
974              
975              StringBuffer message = new StringBuffer("Save File");
976              message.append(" : ").append(EclipseSensor.this.extractFileName(fileResource));
977              
978              keyValueMap.put(EclipseSensorConstants.SUBTYPE, "Save");
979              EclipseSensor.this.addDevEvent(EclipseSensorConstants.DEVEVENT_EDIT, fileResource, 
980                  keyValueMap, message.toString());
981            }
982    
983            // Visit the children because it is not necessary for the saving file to be only one file.
984            return true;
985          }
986          return true; // visit the children
987        }
988      }
